const colors = ["white", "red", "blue", "orange", "gray"];
const btn = document.querySelector("#myBtn");
const body = document.body;
let start = 0;

body.style.backgroundColor = colors[start];

btn.addEventListener("click", () => {
  if (start === colors.length) start = 0;
  if (start < colors.length) {
    body.style.backgroundColor = colors[start];
    start += 1;
  }
});
